\section{Problem \& Motivation}
\begin{frame}{The problem: phishing on Ethereum}
\begin{itemize}
\item On-chain phishing drains \textbf{hundreds of millions of USD} annually (address poisoning, ice-phishing approvals, malicious Seaport/NFT orders).
\item Two questions, usually solved \emph{separately}:
\begin{itemize}
\item \textbf{Account level} --- is this address a scammer?
\item \textbf{Transaction level} --- \emph{which} transaction is the fraudulent one?
\end{itemize}
\item Rarely solved by \textbf{one model}, and rarely with the \textbf{evaluation rigor} needed to trust small, reused crypto datasets.
\end{itemize}
\vfill
\textbf{Our goal (deliberately constrained):} keep the published backbones unchanged, and show how far \emph{careful unification} + an \emph{honest protocol} can go.
\end{frame}

\section{Baselines \& Their Weaknesses}
\begin{frame}{Baseline paper 1 --- BERT4ETH (WWW'23)}
\begin{itemize}
\item Pre-trained Transformer for Ethereum fraud detection.
\item Each transaction $\to$ embeddings: \textbf{counterparty}, \textbf{IN/OUT direction}, \textbf{amount}, \textbf{count}, \textbf{position/time}, then a Transformer encoder.
\item Strong \textbf{account-level} representation; \textbf{no transaction-level localization}.
\end{itemize}
\end{frame}

\begin{frame}{Baseline paper 2 --- C-TMIL (Temporal MIL)}
\begin{itemize}
\item Multiple-Instance Learning view: \textbf{1 account = 1 bag} of transactions.
\item Pipeline: \textbf{TCN} $\to$ \textbf{Gated-Attention MIL} $\to$ \textbf{Bag Aggregation} $\to$ \textbf{MLP}.
\item Attention weights $\Rightarrow$ candidate localization.
\item \textbf{Key limitation:} C-TMIL \emph{hard-masks inbound} transactions and attends to \emph{outbound only}.
\end{itemize}
\end{frame}

\begin{frame}{Weaknesses we target}
\begin{itemize}
\item \textbf{Outbound-only hard mask} throws away inbound signal --- but approve/permit/ice-phishing is triggered \emph{inbound} (the drain is a \textsc{transferFrom}).
\item \textbf{Two granularities, two models} --- no single unified system.
\item \textbf{``Attention = explanation''} assumed, never tested.
\item \textbf{Evaluation gaps:} no hard negatives, no CIs, no OOD, single-label localization GT, contract-mediated cases unlabeled.
\end{itemize}
\end{frame}

\section{Our Direction \& Key Modification}
\begin{frame}{Our direction --- preserve backbones, unify}
\textbf{Constraints honored:}
\begin{itemize}
\item BERT4ETH embeddings (counterparty+IN/OUT+amount+count+position) --- \textbf{unchanged}.
\item TMIL (TCN $\to$ Gated-Attn MIL $\to$ Bag Agg $\to$ MLP) --- \textbf{unchanged}.
\item No new backbone, no new localization head.
\end{itemize}
\vfill
\textbf{Key modification:}
\begin{itemize}
\item Replace C-TMIL's \emph{outbound-only hard mask} with the \emph{existing BERT4ETH IN/OUT embedding}.
\item Let the model \textbf{learn} whether inbound or outbound matters, rather than assume.
\end{itemize}
\end{frame}

\begin{frame}{Three contributions}
\begin{enumerate}
\item \textbf{Genuinely unified, single-weight model:} one shared encoder, two attention heads, \emph{one forward pass} $\to$ account decision + transaction ranking.
\item \textbf{Audited benchmark + contract-mediated relabeling protocol:} trace execution receipts (\textsc{transferFrom}, Seaport \textsc{OrderFulfilled}, internal ETH); clean GT $83.4\% \to 92.7\%$.
\item \textbf{Honest evaluation:} cluster-bootstrap CIs, by-source negatives, activity strata, OOD splits, and an ablation showing attention \emph{can} be trained to localize but is \textbf{subsumed by engineered features} in fusion.
\end{enumerate}
\end{frame}

\section{Dataset}
\begin{frame}{Datasets used}
\begin{itemize}
\item \textbf{Train (in-domain):} BERT4ETH phishing + normal EOA datasets (11{,}136 train bags; vocab 84{,}982 tokens).
\item \textbf{Test --- account level (cross-domain):} PTXPhish cashier/recipient scammer accounts vs.\ benign accounts.
\item \textbf{Test --- transaction level:} PTXPhish transaction hashes.
\item \textbf{Histories} crawled via \textbf{Etherscan}; model trains \emph{only} on in-domain data $\Rightarrow$ PTXPhish is \textbf{zero-shot}.
\end{itemize}
\end{frame}

\begin{frame}{Ground truth \& the relabeling protocol}
\begin{itemize}
\item \textbf{Direct-transfer phishing:} the PTXPhish hash \emph{is} the localization target (in account history) --- 83.4\%.
\item \textbf{Contract-mediated} (Approve, Permit, Seaport, NFT orders): the listed hash is a victim approval / marketplace call. The value movement is in an \emph{execution receipt}.
\item We \textbf{trace} ERC-20/WETH \textsc{transferFrom}, Seaport \textsc{OrderFulfilled}, internal ETH $\to$ attribute the scammer EOA and \textbf{relabel the execution tx}.
\item Clean account-level GT: \textbf{83.4\% $\to$ 92.7\%} (94.8\% excluding proxy-upgrade infra).
\end{itemize}
\end{frame}

\begin{frame}{Bag definition}
\begin{itemize}
\item \textbf{1 Ethereum account = 1 bag.}
\item \textbf{Single fixed window} $L{=}100$ transactions ($>$90\% of scammer EOAs have $<L$ post-anchor tx).
\item A single window avoids the score-aggregation ambiguity of overlapping sliding windows.
\item Positive windows end at the \textbf{detection cutoff} $\Rightarrow$ recency becomes a strong prior (addressed later).
\item We exclude each bag's \emph{final} tx from candidates \& GT for \textbf{all} methods (neutralize trivial last-tx artifact).
\end{itemize}
\end{frame}

\section{Architecture}
\begin{frame}{Unified TMIL architecture}
\centering
\includegraphics[height=0.78\textheight]{../figures/fig_architecture.png}
\end{frame}

\begin{frame}{Architecture --- in words}
\begin{itemize}
\item \textbf{Shared encoder:} BERT4ETH per-tx embeddings (counterparty/IN-OUT/amount/count/time) $\to$ LayerNorm $\to$ \textbf{TCN}.
\item \textbf{Head-C} (soft gated-attention): pools the bag $\to$ \textbf{account} BCE.
\item \textbf{Head-L}: ranks transactions for \textbf{localization}; uses the \emph{IN/OUT embedding} instead of a hard mask, trained \emph{jointly} under BCE.
\item One set of weights, \textbf{one forward pass}.
\item Loss: $\mathcal{L}=\mathrm{BCE}(C)+\lambda\,\mathrm{BCE}(L)+\beta\,\mathcal{L}_{\text{contrast}}$.
\item \textbf{Finding:} a post-hoc read-out from the soft head is \emph{insufficient} --- Head-L must receive gradient.
\end{itemize}
\end{frame}

\begin{frame}{Backbone ablation (account-level transfer)}
\centering\small
\begin{tabular}{lccc}\toprule

Full model & 0.734 & 0.974 & 0.849\\
\ \ $-$ TCN encoder & 0.735 & 0.955 & 0.592\\
\ \ $-$ counterparty emb. & 0.554 & 0.960 & 0.461\\
\ \ $-$ IN/OUT emb. & 0.717 & 0.928 & 0.662\\

$\lambda{=}0.25$ & 0.735 & 0.969 & 0.837\\\bottomrule
\end{tabular}
\vfill
\footnotesize Counterparty + IN/OUT embeddings drive hard-negative transfer $\Rightarrow$ justifies replacing the outbound mask with the IN/OUT embedding. Joint Head-L ($\lambda{>}0$) does not hurt accounts.
\end{frame}

\begin{frame}{Ablation summary (the three asked-for axes)}
\begin{itemize}
\item \textbf{Outbound mask vs.\ IN/OUT embedding:} IN/OUT embedding restores hard-neg AUC ($0.662 \to 0.849$); learned direction $>$ assumed outbound-only.
\item \textbf{Payload features ON/OFF:} engineered on-chain/payload features are what \emph{drive} localization (Hit@1 $0.41 \to 0.83$).
\item \textbf{Attention-only vs.\ feature-assisted localization:} attention-only $0.416\!\to\!0.515$ (trainable) but features set the ceiling in fusion (strong vs no-attn $p{=}0.68$).
\end{itemize}
\end{frame}

\section{Discussion}
\begin{frame}{Discussion \& limitations (stated, not hidden)}
\begin{itemize}
\item \textbf{Small hard-negative pool} (80 DeFi/KOL): non-trivial variance $\Rightarrow$ we report cluster-bootstrap CIs. We keep PTXPhish as-is for comparability.
\item \textbf{Positional bias:} positive windows end at detection cutoff $\Rightarrow$ recency is strong; we neutralize the trivial last-tx and benchmark against recency.
\item \textbf{Attention is not faithful} for localization (our ablation) --- no explanatory claim made.
\item \textbf{Coverage:} clean GT 92.7\%; proxy-upgrade infra left for future tracing.
\item Account model is \emph{competitive, not strictly SOTA --- by design}.
\end{itemize}
\end{frame}

\begin{frame}{Conclusion}
\begin{itemize}
\item \textbf{Unified TMIL:} account + transaction phishing detection in \emph{one forward pass}, preserving BERT4ETH and TMIL.
\item \textbf{Key enabler:} replace the outbound-only hard mask with the learned \textbf{IN/OUT embedding}.
\item \textbf{Benchmark contribution:} audited PTXPhish + contract-mediated relabeling protocol.
\item \textbf{Localization is SOTA on this benchmark:} content-aware ranker beats every MIL/heuristic baseline ($0.832$ vs recency $0.693$, $p{=}0.002$). Attention is a useful but non-essential signal.
\item We release benchmark, protocol, baselines, and evaluation code.
\end{itemize}
