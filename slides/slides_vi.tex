\section{Bài toán \& Động lực}
\begin{frame}{Bài toán: lừa đảo trên Ethereum}
\begin{itemize}
\item Lừa đảo on-chain rút \textbf{hàng trăm triệu USD} mỗi năm (address poisoning, ice-phishing approvals, lệnh Seaport/NFT độc hại).
\item Hai câu hỏi, thường được giải \emph{riêng rẽ}:
\begin{itemize}
\item \textbf{Mức tài khoản} --- địa chỉ này có phải kẻ lừa đảo?
\item \textbf{Mức giao dịch} --- \emph{giao dịch nào} mới là giao dịch gian lận?
\end{itemize}
\item Hiếm khi giải bằng \textbf{một mô hình}, và hiếm khi đủ \textbf{độ chặt chẽ đánh giá} để tin các tập dữ liệu crypto nhỏ, bị tái sử dụng.
\end{itemize}
\vfill
\textbf{Mục tiêu (cố tình ràng buộc):} giữ nguyên backbone đã công bố, và cho thấy \emph{hợp nhất cẩn thận} + một \emph{giao thức trung thực} đi xa đến đâu.
\end{frame}

\section{Baseline \& Điểm yếu}
\begin{frame}{Baseline 1 --- BERT4ETH (WWW'23)}
\begin{itemize}
\item Transformer tiền-huấn-luyện cho phát hiện gian lận Ethereum.
\item Mỗi giao dịch $\to$ embedding: \textbf{đối tác}, \textbf{hướng IN/OUT}, \textbf{số tiền}, \textbf{đếm}, \textbf{vị trí/thời gian}, rồi bộ mã hóa Transformer.
\item Biểu diễn \textbf{mức tài khoản} mạnh; \textbf{không có định vị mức giao dịch}.
\end{itemize}
\end{frame}

\begin{frame}{Baseline 2 --- C-TMIL (Temporal MIL)}
\begin{itemize}
\item Góc nhìn Multiple-Instance Learning: \textbf{1 tài khoản = 1 túi} giao dịch.
\item Pipeline: \textbf{TCN} $\to$ \textbf{Gated-Attention MIL} $\to$ \textbf{Bag Aggregation} $\to$ \textbf{MLP}.
\item Trọng số attention $\Rightarrow$ định vị ứng viên.
\item \textbf{Hạn chế chính:} C-TMIL \emph{che cứng inbound} và chỉ attend \emph{outbound}.
\end{itemize}
\end{frame}

\begin{frame}{Các điểm yếu chúng tôi nhắm tới}
\begin{itemize}
\item \textbf{Mặt nạ cứng chỉ-outbound} vứt bỏ tín hiệu inbound --- nhưng approve/permit/ice-phishing được kích hoạt \emph{từ inbound} (cú rút là một \textsc{transferFrom}).
\item \textbf{Hai mức độ, hai mô hình} --- không có hệ thống hợp nhất duy nhất.
\item \textbf{Giả định ``attention = giải thích''} --- chưa bao giờ được kiểm chứng.
\item \textbf{Lỗ hổng đánh giá:} thiếu mẫu âm khó, thiếu CI, thiếu OOD, GT định vị đơn nhãn, các ca qua hợp đồng chưa gán nhãn.
\end{itemize}
\end{frame}

\section{Hướng đi \& Thay đổi cốt lõi}
\begin{frame}{Hướng đi --- giữ backbone, hợp nhất}
\textbf{Ràng buộc tuân thủ:}
\begin{itemize}
\item Embedding BERT4ETH (đối-tác+IN/OUT+số-tiền+đếm+vị-trí) --- \textbf{không đổi}.
\item TMIL (TCN $\to$ Gated-Attn MIL $\to$ Bag Agg $\to$ MLP) --- \textbf{không đổi}.
\item Không backbone mới, không đầu định vị mới.
\end{itemize}
\vfill
\textbf{Thay đổi cốt lõi:}
\begin{itemize}
\item Thay \emph{mặt nạ cứng chỉ-outbound} của C-TMIL bằng \emph{embedding IN/OUT sẵn có} của BERT4ETH.
\item Để mô hình \textbf{tự học} inbound hay outbound mới quan trọng, thay vì giả định.
\end{itemize}
\end{frame}

\begin{frame}{Ba đóng góp}
\begin{enumerate}
\item \textbf{Mô hình hợp nhất thực sự, một-bộ-trọng-số:} một bộ mã hóa dùng chung, hai đầu attention, \emph{một lượt truyền xuôi} $\to$ quyết định tài khoản + xếp hạng giao dịch.
\item \textbf{Benchmark kiểm toán + giao thức gán nhãn lại qua hợp đồng:} truy vết biên lai thực thi (\textsc{transferFrom}, Seaport \textsc{OrderFulfilled}, ETH nội bộ); GT sạch $83.4\% \to 92.7\%$.
\item \textbf{Đánh giá trung thực:} CI bootstrap cụm, âm theo nguồn, phân tầng hoạt động, phép chia OOD, và ablation cho thấy attention \emph{có thể} được huấn luyện để định vị nhưng bị \textbf{đặc trưng thủ công thâu tóm} trong hợp nhất.
\end{enumerate}
\end{frame}

\section{Dataset}
\begin{frame}{Các tập dữ liệu}
\begin{itemize}
\item \textbf{Huấn luyện (nội miền):} tập phishing + EOA bình thường của BERT4ETH (11.136 túi train; vocab 84.982 token).
\item \textbf{Kiểm thử --- mức tài khoản (chéo miền):} tài khoản cashier/recipient kẻ lừa đảo PTXPhish so với tài khoản lành tính.
\item \textbf{Kiểm thử --- mức giao dịch:} các hash giao dịch PTXPhish.
\item \textbf{Lịch sử} crawl qua \textbf{Etherscan}; mô hình chỉ huấn luyện trên dữ liệu nội miền $\Rightarrow$ PTXPhish là \textbf{zero-shot}.
\end{itemize}
\end{frame}

\begin{frame}{Ground truth \& giao thức gán nhãn lại}
\begin{itemize}
\item \textbf{Phishing chuyển trực tiếp:} hash PTXPhish \emph{chính là} mục tiêu định vị (nằm trong lịch sử) --- 83.4\%.
\item \textbf{Qua hợp đồng} (Approve, Permit, Seaport, NFT order): hash được liệt kê là một phê duyệt nạn nhân / lời gọi sàn. Dịch chuyển giá trị nằm trong một \emph{biên lai thực thi}.
\item Chúng tôi \textbf{truy vết} \textsc{transferFrom} ERC-20/WETH, Seaport \textsc{OrderFulfilled}, ETH nội bộ $\to$ quy kết EOA kẻ lừa đảo và \textbf{gán nhãn lại giao dịch thực thi}.
\item GT mức tài khoản sạch: \textbf{83.4\% $\to$ 92.7\%} (94.8\% nếu loại hạ tầng nâng cấp proxy).
\end{itemize}
\end{frame}

\begin{frame}{Định nghĩa túi (bag)}
\begin{itemize}
\item \textbf{1 tài khoản Ethereum = 1 túi.}
\item \textbf{Cửa sổ cố định} $L{=}100$ giao dịch ($>$90\% EOA kẻ lừa đảo có $<L$ tx sau mốc neo).
\item Cửa sổ đơn tránh sự mơ hồ tổng hợp điểm của các cửa sổ trượt chồng lấn.
\item Cửa sổ dương kết thúc tại \textbf{mốc phát hiện} $\Rightarrow$ recency thành prior mạnh (xử lý sau).
\item Loại giao dịch \emph{cuối} của mỗi túi khỏi ứng viên \& GT cho \textbf{mọi} phương pháp (trung hòa tạo tác tx-cuối tầm thường).
\end{itemize}
\end{frame}

\begin{frame}{Kiến trúc --- diễn giải}
\begin{itemize}
\item \textbf{Bộ mã hóa dùng chung:} embedding BERT4ETH mỗi tx (đối-tác/IN-OUT/số-tiền/đếm/thời-gian) $\to$ LayerNorm $\to$ \textbf{TCN}.
\item \textbf{Head-C} (gated-attention mềm): gộp túi $\to$ BCE \textbf{tài khoản}.
\item \textbf{Head-L}: xếp hạng giao dịch để \textbf{định vị}; dùng \emph{embedding IN/OUT} thay mặt nạ cứng, huấn luyện \emph{đồng thời} dưới BCE.
\item Một bộ trọng số, \textbf{một lượt truyền xuôi}.
\item Mất mát: $\mathcal{L}=\mathrm{BCE}(C)+\lambda\,\mathrm{BCE}(L)+\beta\,\mathcal{L}_{\text{contrast}}$.
\item \textbf{Phát hiện:} đọc hậu kỳ từ đầu mềm là \emph{không đủ} --- Head-L phải nhận gradient.
\end{itemize}
\end{frame}

Mô hình đầy đủ & 0.734 & 0.974 & 0.849\\
\ \ $-$ bộ mã hóa TCN & 0.735 & 0.955 & 0.592\\
\ \ $-$ embedding đối tác & 0.554 & 0.960 & 0.461\\
\ \ $-$ embedding IN/OUT & 0.717 & 0.928 & 0.662\\

$\lambda{=}0.25$ & 0.735 & 0.969 & 0.837\\\bottomrule

\vfill
\footnotesize Embedding đối tác + IN/OUT dẫn dắt chuyển giao âm-khó $\Rightarrow$ biện minh thay mặt nạ outbound bằng embedding IN/OUT. Head-L đồng thời ($\lambda{>}0$) không hại tài khoản.
\end{frame}

\begin{frame}{Tóm tắt ablation (ba trục được yêu cầu)}
\begin{itemize}
\item \textbf{Mặt nạ outbound vs.\ embedding IN/OUT:} embedding IN/OUT khôi phục AUC âm-khó ($0.662 \to 0.849$); hướng học được $>$ chỉ-outbound giả định.
\item \textbf{Đặc trưng payload BẬT/TẮT:} đặc trưng on-chain/payload thủ công mới \emph{dẫn dắt} định vị (Hit@1 $0.41 \to 0.83$).
\item \textbf{Chỉ-attention vs.\ có trợ giúp đặc trưng:} chỉ-attention $0.416\!\to\!0.515$ (huấn luyện được) nhưng đặc trưng đặt trần trong hợp nhất (mạnh vs không-attn $p{=}0.68$).
\end{itemize}
\end{frame}

\section{Thảo luận}
\begin{frame}{Thảo luận \& hạn chế (nêu rõ, không giấu)}
\begin{itemize}
\item \textbf{Bể mẫu âm khó nhỏ} (80 DeFi/KOL): phương sai không nhỏ $\Rightarrow$ báo cáo CI bootstrap cụm. Giữ nguyên PTXPhish để so sánh được.
\item \textbf{Thiên lệch vị trí:} cửa sổ dương kết thúc tại mốc phát hiện $\Rightarrow$ recency mạnh; trung hòa tx-cuối tầm thường và đo so với recency.
\item \textbf{Attention là tín hiệu hữu ích nhưng không thiết yếu} cho định vị; đặc trưng thủ công đặt trần (ablation của chúng tôi).
\item \textbf{Độ phủ:} GT sạch 92.7\%; hạ tầng nâng cấp proxy để dành truy vết tương lai.
\item Mô hình tài khoản \emph{cạnh tranh, không phải SOTA tuyệt đối --- theo thiết kế}.
\end{itemize}
\end{frame}

\begin{frame}{Kết luận}
\begin{itemize}
\item \textbf{Unified TMIL:} phát hiện phishing tài khoản + giao dịch trong \emph{một lượt truyền xuôi}, giữ nguyên BERT4ETH và TMIL.
\item \textbf{Yếu tố mở đường:} thay mặt nạ cứng chỉ-outbound bằng \textbf{embedding IN/OUT} học được.
\item \textbf{Đóng góp benchmark:} PTXPhish kiểm toán + giao thức gán nhãn lại qua hợp đồng.
\item \textbf{Định vị đạt SOTA trên benchmark này:} bộ xếp hạng nhận-biết-nội-dung thắng mọi baseline MIL/heuristic ($0.832$ vs recency $0.693$, $p{=}0.002$). Attention hữu ích nhưng không thiết yếu.
\item Chúng tôi công bố benchmark, giao thức, baseline và mã đánh giá.
\end{itemize}

\end{frame}
